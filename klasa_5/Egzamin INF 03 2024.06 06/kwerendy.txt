-- Zapytanie 1
SELECT imie, nazwisko, koszt 
FROM uczestnicy 
WHERE zaplacono = 0;

-- Zapytanie 2
SELECT AVG(koszt) AS 'Średni koszt', 
       SUM(koszt) AS 'Całkowity koszt', 
       COUNT(*) AS 'Liczba uczestników' 
FROM uczestnicy;

-- Zapytanie 3
SELECT w.nazwa, u.nazwisko, u.email 
FROM wyklady w 
JOIN uczestnicy u ON w.id = u.wyklady_id 
WHERE u.zaplacono = 1;
